Using simplerCo


var fs = require('fs');

var readFile = function() {
  var args = Array.prototype.slice.call(arguments,0);
  return function(callback) {
    args.push(callback);
    return fs.readFile.apply(null, args);
  };
};

simplerCo(function*() {
  var content = yield readFile('code/a.txt', 'utf8');
  console.log(content);
});
